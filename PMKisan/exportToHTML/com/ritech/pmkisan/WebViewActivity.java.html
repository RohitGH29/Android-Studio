9    import android.widget.Toast;
10   
11   import androidx.appcompat.app.AppCompatActivity;
12   
13   import com.firebase.client.DataSnapshot;
14   import com.firebase.client.Firebase;
15   import com.firebase.client.FirebaseError;
16   import com.firebase.client.ValueEventListener;
17   import com.google.android.gms.ads.AdRequest;
18   import com.google.android.gms.ads.AdSize;
19   import com.google.android.gms.ads.AdView;
20   import com.google.android.gms.ads.MobileAds;
21   import com.google.android.gms.ads.initialization.InitializationStatus;
22   import com.google.android.gms.ads.initialization.OnInitializationCompleteListener;
23   
24   public class WebViewActivity extends AppCompatActivity {
25   
26       private WebView webView;
27       RelativeLayout banner;
28       String data;
29   
30       @Override
31       protected void onCreate(Bundle savedInstanceState) {
32           super.onCreate(savedInstanceState);
33           setContentView(R.layout.activity_web_view);
34   
35           getSupportActionBar().hide();
36   
37           Intent intent= getIntent();
38           String url= intent.getStringExtra("url");
39   
40   
41           webView = (WebView) findViewById(R.id.webView);
42           webView.setWebViewClient(new WebViewClient());
43           webView.loadUrl(url);
44   
45           MobileAds.initialize(this, new OnInitializationCompleteListener() {
46               @Override
47               public void onInitializationComplete(InitializationStatus initializationStatus) {
48               }
49           });
50   
51           banner=findViewById(R.id.banner_container);
52           fireAds("https://pm-kisan-ff3d4-default-rtdb.firebaseio.com/Banner");
53   
54       }
55   
56       public void fireAds(String adsUrl) {
57           Firebase.setAndroidContext(this);
58           Firebase firebase = new Firebase(adsUrl);
59   
60           firebase.addValueEventListener(new ValueEventListener() {
61               @Override
62               public void onDataChange(DataSnapshot dataSnapshot) {
63                   data = dataSnapshot.getValue(String.class);
64                   AdView mAdView = new AdView(WebViewActivity.this);
65                   mAdView.setAdUnitId(data);
66                   banner.addView(mAdView);
67                   mAdView.setAdSize(AdSize.FULL_BANNER);
68                   AdRequest adRequest = new AdRequest.Builder().build();
69                   mAdView.loadAd(adRequest);
70               }
71   
72               @Override
73               public void onCancelled(FirebaseError firebaseError) {
74                   Toast.makeText(WebViewActivity.this, "Failed", Toast.LENGTH_SHORT).show();
75   
76               }
77           });
78       }
79   
80   
81       @Override
82       public void onBackPressed() {
83           if (webView.canGoBack()){
84               webView.goBack();
85           }else {
86               super.onBackPressed();
87           }
88       }
89   }
